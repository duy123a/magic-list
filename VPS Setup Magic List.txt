    keyexchange=ikev1
    authby=secret
    ike=aes256-sha1-modp1024,aes128-sha1-modp1024,3des-sha1-modp1024
    esp=aes256-sha1,aes128-sha1,3des-sha1
    type=transport
    left=%defaultroute
    leftprotoport=17/1701
    right=118.70.67.199
    rightprotoport=17/1701

[lac myvpn]
